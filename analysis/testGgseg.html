Random data

someData <- tibble(
  region = rep(c("parahippocampal", "medial orbitofrontal"), 2), 
  p = sample(seq(0,.5,.001), 4),
  groups = c(rep("g1", 2), rep("g2", 2))
)

Example visualization:

someData %>%
  group_by(groups) %>%
  ggplot() +
  geom_brain(atlas = dk, 
             position = position_brain(hemi ~ side),
             aes(fill = p)) +
  facet_wrap(~groups) + scale_fill_viridis() +theme_minimal()
## merging atlas and data by 'region'
